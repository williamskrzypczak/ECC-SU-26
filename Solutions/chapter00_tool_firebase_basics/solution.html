Security Note

Test mode is for learning only. It allows anyone to read and write your database for a limited time. A public app needs proper Firestore security rules and usually Firebase Authentication before release.

ContentView.swift

import SwiftUI
import FirebaseFirestore

struct Announcement: Identifiable {
    let id: String
    let title: String
    let author: String
}

struct ContentView: View {
    @State private var titleText = ""
    @State private var announcements: [Announcement] = []
    @State private var statusMessage = "Ready"

    private let db = Firestore.firestore()

    var body: some View {
        NavigationStack {
            VStack(spacing: 16) {
                TextField("Announcement title", text: $titleText)
                    .textFieldStyle(.roundedBorder)

                Button("Post") {
                    postAnnouncement()
                }
                .buttonStyle(.borderedProminent)

                Text(statusMessage)
                    .font(.footnote)
                    .foregroundColor(.secondary)

                List(announcements) { item in
                    VStack(alignment: .leading, spacing: 4) {
                        Text(item.title)
                            .font(.headline)
                        Text("By \(item.author)")
                            .font(.subheadline)
                            .foregroundColor(.secondary)
                    }
                }
            }
            .padding()
            .navigationTitle("Campus Announcements")
            .onAppear {
                loadAnnouncements()
            }
        }
    }

    private func postAnnouncement() {
        let trimmed = titleText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !trimmed.isEmpty else {
            statusMessage = "Enter a title first."
            return
        }

        db.collection("announcements").addDocument(data: [
            "title": trimmed,
            "author": "Your Name",
            "createdAt": Timestamp(date: Date())
        ]) { error in
            if let error = error {
                statusMessage = "Write failed: \(error.localizedDescription)"
                return
            }
            titleText = ""
            statusMessage = "Posted to Firestore."
            loadAnnouncements()
        }
    }

    private func loadAnnouncements() {
        db.collection("announcements")
            .order(by: "createdAt", descending: true)
            .getDocuments { snapshot, error in
                if let error = error {
                    statusMessage = "Read failed: \(error.localizedDescription)"
                    return
                }

                announcements = snapshot?.documents.compactMap { doc in
                    let data = doc.data()
                    guard let title = data["title"] as? String,
                          let author = data["author"] as? String else {
                        return nil
                    }
                    return Announcement(id: doc.documentID, title: title, author: author)
                } ?? []
            }
    }
}

Sample Reflection Answers

Collection vs. document: A collection is a group of documents, like a sheet tab. A document is one record inside that collection, like a single row.

GoogleService-Info.plist: It connects your iOS app to the correct Firebase project so the SDK knows where to read and write data.

Test mode: Test mode lets anyone read and write your database, which is unsafe for a public app because strangers could delete or change your data.
